<template>
  <div class="app">
    <h1>Custom Grid Demo</h1>
    
    <!-- カスタムグリッドの呼び出し -->
    <custom-grid
      :data="gridData"
      :columns="gridColumns"
      :grid-width="800"
      :grid-height="600"
      :visible-row-count="15"
      ref="customGrid"
    ></custom-grid>

    <!-- ボタンで選択された行のデータを表示する例 -->
    <div>
      <h2>Selected Rows</h2>
      <ul>
        <li v-for="row in selectedRows" :key="row.id">
          {{ row.name }} - {{ row.age }} years old
        </li>
      </ul>
    </div>

    <!-- ボタンで全行のデータを表示する例 -->
    <div>
      <h2>All Rows</h2>
      <ul>
        <li v-for="row in allRows" :key="row.id">
          {{ row.name }} - {{ row.age }} years old
        </li>
      </ul>
    </div>
  </div>
</template>

<script>
import CustomGrid from './components/CustomGrid.vue';

export default {
  name: 'App',
  components: {
    CustomGrid,
  },
  data() {
    return {
      gridData: [
        { id: 1, name: 'Alice', age: 25 },
        { id: 2, name: 'Bob', age: 30 },
        { id: 3, name: 'Charlie', age: 28 },
        { id: 4, name: 'David', age: 35 },
        // 他のデータも追加可能
      ],
      gridColumns: [
        { label: 'ID', field: 'id', width: 50 },
        { label: 'Name', field: 'name', width: 150 },
        { label: 'Age', field: 'age', width: 100 },
        // 他のカラムも追加可能
      ],
      selectedRows: [],
      allRows: []
    };
  },
  methods: {
    getSelectedRowsFromCustomGrid() {
      this.selectedRows = this.$refs.customGrid.getSelectedRows();
    },
    getAllRowsFromCustomGrid() {
      this.allRows = this.$refs.customGrid.getAllRows();
    }
  }
};
</script>

<style>
.app {
  max-width: 800px;
  margin: 0 auto;
  padding: 20px;
}
</style>
